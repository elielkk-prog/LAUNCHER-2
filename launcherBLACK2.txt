<script>
    // Ícones disponíveis
    const EMOJIS = ['🔍', '🌐', '📧', '📱', '💼', '🎮', '🎵', '📺', '🎥', '📷', '🛍️', '📚', '⚙️', '🔐', '🎨', '📊', '💰', '🏃', '🍔', '☁️'];
    
    // Estado da aplicação
    let links = JSON.parse(localStorage.getItem('minhaLauncher')) || [];
    let categorias = JSON.parse(localStorage.getItem('categorias')) || ['Favoritos', 'Trabalho', 'Diversão', 'Educação', 'Ferramentas'];
    let categoriaAtiva = 'Todos';
    let estaEmModoExcluir = false;
    let editandoIndex = null;

    // Inicialização
    function init() {
        popularEmojis();
        renderizarCategorias();
        renderizar();
        buscarClima();
        atualizarHora();
        setInterval(atualizarHora, 1000);
        
        // Atalho de teclado: "/" para focar no search
        document.addEventListener('keydown', (e) => {
            if (e.key === '/' && document.activeElement !== document.getElementById('pesquisa')) {
                e.preventDefault();
                document.getElementById('pesquisa').focus();
            }
        });
        
        // Enter no search
        document.getElementById('pesquisa').addEventListener('keydown', (e) => {
            if (e.key === 'Enter') buscarWeb();
        });
    }

    function popularEmojis() {
        const picker = document.getElementById('iconPicker');
        const editPicker = document.getElementById('editIconPicker');
        picker.innerHTML = '';
        editPicker.innerHTML = '';
        
        EMOJIS.forEach(emoji => {
            const div = document.createElement('div');
            div.className = 'icon-option';
            div.textContent = emoji;
            div.onclick = () => selecionarEmoji('novoModal', emoji);
            picker.appendChild(div);
            
            const editDiv = div.cloneNode(true);
            editDiv.onclick = () => selecionarEmoji('editModal', emoji);
            editPicker.appendChild(editDiv);
        });
    }

    function selecionarEmoji(modal, emoji) {
        const inputId = modal === 'novoModal' ? 'customIcon' : 'editCustomIcon';
        document.getElementById(inputId).value = emoji;
        document.querySelectorAll(`#${modal} .icon-option`).forEach(el => el.classList.remove('selected'));
        event.target.classList.add('selected');
    }

    function renderizarCategorias() {
        const container = document.getElementById('categoryTabs');
        container.innerHTML = '';
        
        const todosBtn = document.createElement('button');
        todosBtn.className = 'category-tab' + (categoriaAtiva === 'Todos' ? ' active' : '');
        todosBtn.textContent = 'Todos';
        todosBtn.onclick = () => {
            categoriaAtiva = 'Todos';
            document.querySelectorAll('.category-tab').forEach(t => t.classList.remove('active'));
            todosBtn.classList.add('active');
            renderizar();
        };
        container.appendChild(todosBtn);
        
        categorias.forEach(cat => {
            const btn = document.createElement('button');
            btn.className = 'category-tab' + (categoriaAtiva === cat ? ' active' : '');
            btn.textContent = cat;
            btn.onclick = () => {
                categoriaAtiva = cat;
                document.querySelectorAll('.category-tab').forEach(t => t.classList.remove('active'));
                btn.classList.add('active');
                renderizar();
            };
            container.appendChild(btn);
        });
    }

    function renderizar() {
        const container = document.getElementById("tiles");
        container.innerHTML = '';
        
        // Botão de adicionar
        const addTile = document.createElement("div");
        addTile.className = "tile add-tile";
        addTile.textContent = "+";
        addTile.onclick = novoAtalho;
        container.appendChild(addTile);
        
        // Filtrar links
        const linksFiltrados = categoriaAtiva === 'Todos' 
            ? links 
            : links.filter(l => l.categoria === categoriaAtiva);
        
        linksFiltrados.forEach((l, index) => {
            const realIndex = links.indexOf(l);
            const div = document.createElement("div");
            div.className = "tile" + (estaEmModoExcluir ? " modo-excluir" : "");
            
            let iconHtml = '';
            if (l.icon && l.icon.length <= 2) {
                iconHtml = `<div class="emoji-icon">${l.icon}</div>`;
            } else {
                const url = new URL(l.url).hostname;
                iconHtml = `<img src="https://www.google.com/s2/favicons?domain=${url}&sz=64" onerror="this.style.display='none'">`;
            }
            
            div.innerHTML = `
                ${estaEmModoExcluir ? '' : `<button class="edit-btn" onclick="abrirEdicao(${realIndex})">✎</button>`}
                ${iconHtml}
                <span>${l.name}</span>
            `;
            
            div.onclick = (e) => {
                if (e.target.classList.contains('edit-btn') || estaEmModoExcluir === false && e.target.closest('.edit-btn')) return;
                if (estaEmModoExcluir) {
                    if (confirm(`Remover "${l.name}"?`)) {
                        links.splice(realIndex, 1);
                        salvar();
                    }
                } else {
                    window.open(l.url, "_blank");
                }
            };
            
            container.appendChild(div);
        });
    }

    function novoAtalho() {
        editandoIndex = null;
        document.getElementById('novoNome').value = '';
        document.getElementById('novaUrl').value = '';
        document.getElementById('customIcon').value = '';
        document.getElementById('novaCategoria').value = categorias[0] || 'Favoritos';
        abrirModal('novoModal');
    }

    function salvarNovo() {
        const n = document.getElementById('novoNome').value.trim();
        const u = document.getElementById('novaUrl').value.trim();
        const icon = document.getElementById('customIcon').value.trim();
        const cat = document.getElementById('novaCategoria').value;
        
        if (!n || !u) {
            mostrarToast('Preencha nome e URL');
            return;
        }
        
        if (!u.startsWith('http')) {
            mostrarToast('URL deve começar com http ou https');
            return;
        }
        
        links.push({ name: n, url: u, icon: icon, categoria: cat });
        fecharModal('novoModal');
        salvar();
        mostrarToast('Atalho adicionado!');
    }

    function abrirEdicao(index) {
        editandoIndex = index;
        const l = links[index];
        document.getElementById('editNome').value = l.name;
        document.getElementById('editUrl').value = l.url;
        document.getElementById('editCustomIcon').value = l.icon || '';
        document.getElementById('editCategoria').value = l.categoria || categorias[0];
        abrirModal('editModal');
    }

    function salvarEdicao() {
        const n = document.getElementById('editNome').value.trim();
        const u = document.getElementById('editUrl').value.trim();
        const icon = document.getElementById('editCustomIcon').value.trim();
        const cat = document.getElementById('editCategoria').value;
        
        if (!n || !u) {
            mostrarToast('Preencha nome e URL');
            return;
        }
        
        if (!u.startsWith('http')) {
            mostrarToast('URL deve começar com http ou https');
            return;
        }
        
        links[editandoIndex] = { name: n, url: u, icon: icon, categoria: cat };
        fecharModal('editModal');
        salvar();
        mostrarToast('Atalho atualizado!');
    }

    function abrirModal(id) {
        document.getElementById(id).classList.add('active');
    }

    function fecharModal(id) {
        document.getElementById(id).classList.remove('active');
    }

    function ordenarLinks() {
        links.sort((a, b) => a.name.localeCompare(b.name, 'pt-BR', { sensitivity: 'base' }));
    }

    function salvar() {
        ordenarLinks();
        localStorage.setItem('minhaLauncher', JSON.stringify(links));
        localStorage.setItem('categorias', JSON.stringify(categorias));
        renderizar();
    }

    function alternarModo() {
        estaEmModoExcluir = !estaEmModoExcluir;
        document.getElementById('btnModoExcluir').classList.toggle('delete-mode', estaEmModoExcluir);
        renderizar();
    }

    function buscarWeb() {
        const query = document.getElementById('pesquisa').value.trim();
        if (query) {
            window.open(`https://www.google.com/search?q=${encodeURIComponent(query)}`);
            document.getElementById('pesquisa').value = '';
        }
    }

    function atualizarHora() {
        const agora = new Date();
        document.getElementById('hora').textContent = agora.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit', second: '2-digit' });
        
        const opcoes = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        const dataFormatada = agora.toLocaleDateString('pt-BR', opcoes);
        document.getElementById('data-completa').textContent = dataFormatada;
    }

    async function buscarClima() {
        const el = document.getElementById('temperatura');
        const buscar = async (lat, lon) => {
            try {
                const res = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,weather_code`);
                const data = await res.json();
                const temp = Math.round(data.current.temperature_2m);
                el.textContent = `📍 ${temp}°C`;
            } catch {
                el.textContent = '🌍 Clima indisponível';
            }
        };
        
        if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(
                (pos) => buscar(pos.coords.latitude, pos.coords.longitude),
                () => buscar(-25.4284, -49.2733) // Curitiba fallback
            );
        } else {
            buscar(-25.4284, -49.2733);
        }
    }

    function exportarBackup() {
        const backup = {
            links: links,
            categorias: categorias,
            exportedAt: new Date().toISOString()
        };
        const blob = new Blob([JSON.stringify(backup, null, 2)], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `launcher-backup-${new Date().toISOString().split('T')[0]}.json`;
        a.click();
        URL.revokeObjectURL(url);
        mostrarToast('Backup exportado!');
    }

    function importarBackup(e) {
        const file = e.target.files[0];
        if (!file) return;
        
        const reader = new FileReader();
        reader.onload = (event) => {
            try {
                const data = JSON.parse(event.target.result);
                
                if (data.links && Array.isArray(data.links)) {
                    links = data.links;
                    if (data.categorias && Array.isArray(data.categorias)) {
                        categorias = data.categorias;
                    }
                    salvar();
                    renderizarCategorias();
                    mostrarToast('Backup importado com sucesso!');
                } else {
                    mostrarToast('Formato de arquivo inválido!');
                }
            } catch {
                mostrarToast('Erro ao importar arquivo!');
            }
        };
        reader.readAsText(file);
    }

    function mostrarToast(msg) {
        const toast = document.createElement('div');
        toast.className = 'toast';
        toast.textContent = msg;
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 2000);
    }

    // Popular dropdowns
    function atualizarDropdowns() {
        const novaCategoria = document.getElementById('novaCategoria');
        const editCategoria = document.getElementById('editCategoria');
        
        novaCategoria.innerHTML = categorias.map(cat => `<option value="${cat}">${cat}</option>`).join('');
        editCategoria.innerHTML = categorias.map(cat => `<option value="${cat}">${cat}</option>`).join('');
    }

    // Atalho: Ctrl+N para novo
    document.addEventListener('keydown', (e) => {
        if ((e.ctrlKey || e.metaKey) && e.key === 'n') {
            e.preventDefault();
            novoAtalho();
        }
    });

    // Iniciar
    atualizarDropdowns();
    init();
